CREATE TABLE IF NOT EXISTS jornadas (
  id_jornada INT NOT NULL AUTO_INCREMENT,
  nombre VARCHAR(100) NOT NULL,
  descripcion VARCHAR(255) DEFAULT NULL,
  PRIMARY KEY (id_jornada),
  UNIQUE KEY ux_jornadas_nombre (nombre)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS jornadas_sedes (
  id INT NOT NULL AUTO_INCREMENT,
  id_jornada INT NOT NULL,
  id_sede INT NOT NULL,
  PRIMARY KEY (id),
  UNIQUE KEY ux_jornada_sede (id_jornada, id_sede),
  KEY (id_jornada),
  KEY (id_sede),
  CONSTRAINT fk_jornadasedes_jornada FOREIGN KEY (id_jornada) REFERENCES jornadas(id_jornada),
  CONSTRAINT fk_jornadasedes_sede FOREIGN KEY (id_sede) REFERENCES sedes(id_sede)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS salones (
  id_salon INT NOT NULL AUTO_INCREMENT,
  nombre VARCHAR(100) NOT NULL,
  codigo VARCHAR(50) DEFAULT NULL,
  id_sede INT DEFAULT NULL,
  ubicacion VARCHAR(255) DEFAULT NULL,
  descripcion TEXT DEFAULT NULL,
  PRIMARY KEY (id_salon),
  KEY (id_sede),
  CONSTRAINT fk_salones_sedes FOREIGN KEY (id_sede) REFERENCES sedes(id_sede)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS camaras (
  cam_id VARCHAR(50) PRIMARY KEY,
  nombre VARCHAR(120) DEFAULT NULL,
  source VARCHAR(255) DEFAULT NULL,
  id_sede INT DEFAULT NULL,
  descripcion TEXT DEFAULT NULL,
  CONSTRAINT fk_camaras_sedes FOREIGN KEY (id_sede) REFERENCES sedes(id_sede)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS camera_mappings (
  id INT NOT NULL AUTO_INCREMENT,
  cam_id VARCHAR(50) NOT NULL,
  id_salon INT DEFAULT NULL,
  id_sede_carrera INT DEFAULT NULL,
  id_seccion INT DEFAULT NULL,
  id_jornada INT DEFAULT NULL,
  activo TINYINT(1) DEFAULT 1,
  PRIMARY KEY (id),
  UNIQUE KEY ux_cammap (cam_id, id_salon, id_sede_carrera, id_seccion, id_jornada),
  KEY (cam_id),
  KEY (id_salon),
  KEY (id_sede_carrera),
  KEY (id_seccion),
  KEY (id_jornada),
  CONSTRAINT fk_cammap_cam FOREIGN KEY (cam_id) REFERENCES camaras(cam_id),
  CONSTRAINT fk_cammap_sede_carrera FOREIGN KEY (id_sede_carrera) REFERENCES sede_carrera(id_sede_carrera),
  CONSTRAINT fk_cammap_seccion FOREIGN KEY (id_seccion) REFERENCES secciones(id_seccion),
  CONSTRAINT fk_cammap_salon FOREIGN KEY (id_salon) REFERENCES salones(id_salon),
  CONSTRAINT fk_cammap_jornada FOREIGN KEY (id_jornada) REFERENCES jornadas(id_jornada)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS horarios (
  id_horario INT NOT NULL AUTO_INCREMENT,
  id_asignacion INT NOT NULL,
  dia ENUM('LUN','MAR','MIE','JUE','VIE','SAB','DOM') NOT NULL,
  hora_inicio TIME NOT NULL,
  hora_fin TIME NOT NULL,
  id_salon INT DEFAULT NULL,
  PRIMARY KEY (id_horario),
  KEY (id_asignacion),
  KEY (id_salon),
  CONSTRAINT fk_horario_asignacion FOREIGN KEY (id_asignacion) REFERENCES asignacion_cursos(id_asignacion),
  CONSTRAINT fk_horario_salon FOREIGN KEY (id_salon) REFERENCES salones(id_salon)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Insertar horarios: usa columna `dia` si existe, sino `dia_semana` (compatible con volcado previo)
SELECT COUNT(*) INTO @has_dia_col FROM information_schema.COLUMNS
  WHERE TABLE_SCHEMA = DATABASE() AND TABLE_NAME = 'horarios' AND COLUMN_NAME = 'dia';
SET @target_col = IF(@has_dia_col > 0, 'dia', 'dia_semana');
SELECT COLUMN_TYPE INTO @coltype FROM information_schema.COLUMNS
  WHERE TABLE_SCHEMA = DATABASE() AND TABLE_NAME = 'horarios' AND COLUMN_NAME = @target_col LIMIT 1;

SET @val_sab = IF(LOCATE("'SAB'", @coltype) > 0, 'SAB', IF(LOCATE("'sabado'", LOWER(@coltype)) > 0, 'sabado', 'SAB'));
SET @val_dom = IF(LOCATE("'DOM'", @coltype) > 0, 'DOM', IF(LOCATE("'domingo'", LOWER(@coltype)) > 0, 'domingo', 'DOM'));

SET @has_asig = (SELECT COUNT(*) FROM asignacion_cursos WHERE id_asignacion = @id_asig_sab);
SET @ins = IF(@has_asig > 0,
  CONCAT('INSERT INTO horarios (id_asignacion, ', @target_col, ', hora_inicio, hora_fin, id_salon, id_jornada) SELECT ', @id_asig_sab, ', ', QUOTE(@val_sab), ', ''08:00:00'', ''10:00:00'', ', @id_salon_306, ', ', IFNULL(@id_jornada_sab, 'NULL'), ' WHERE @id_jornada_sab IS NOT NULL AND NOT EXISTS (SELECT 1 FROM horarios WHERE id_asignacion = ', @id_asig_sab, ' AND ', @target_col, '=', QUOTE(@val_sab), ' LIMIT 1)')
  , 'SELECT 0');
PREPARE stmt FROM @ins; EXECUTE stmt; DEALLOCATE PREPARE stmt;

SET @has_asig = (SELECT COUNT(*) FROM asignacion_cursos WHERE id_asignacion = @id_asig_dom);

  , 'SELECT 0');
